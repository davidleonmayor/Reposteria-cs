BASE_URL="http://localhost:5232"
# (opcional) ver que Swagger responde
curl -s -o /dev/null -w "swagger http_code=%{http_code}\n" "$BASE_URL/swagger/index.html"
# login (Admin) -> devuelve JSON con token
curl -s -i -X POST "$BASE_URL/api/Auth/login" \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123!"}'
# guardar token
TOKEN=$(curl -s -X POST "$BASE_URL/api/Auth/login" \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123!"}' | jq -r '.token')
echo "TOKEN=$TOKEN"
# CREATE (requiere Admin / PeopleWrite)
curl -s -i -X POST "$BASE_URL/api/PersonType" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"name":"Cliente","description":"Compra productos"}'
# (opcional) capturar el ID creado
ID=$(curl -s -X POST "$BASE_URL/api/PersonType" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"name":"Proveedor","description":"Suministra insumos"}' | jq -r '.id')
echo "ID=$ID"
# READ ALL (anonimo)
curl -s -i "$BASE_URL/api/PersonType"
# READ BY ID (anonimo)
curl -s -i "$BASE_URL/api/PersonType/$ID"
# UPDATE (requiere Admin / PeopleWrite)
curl -s -i -X PUT "$BASE_URL/api/PersonType/$ID" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"name":"Proveedor","description":"Actualizado"}'
# DELETE (requiere Admin / PeopleWrite)
curl -s -i -X DELETE "$BASE_URL/api/PersonType/$ID" \
  -H "Authorization: Bearer $TOKEN"
